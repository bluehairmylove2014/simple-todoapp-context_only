
  const handleChange = (value: string) => {
    dispatch({
      type: "SET_TASK_ACTION",
      payload:
        value.length === 0
          ? MOCK_TASK_DATA
          : state.taskList.filter(
              (task) =>
                value.toLowerCase().includes(task.name.toLowerCase()) ||
                task.name.toLowerCase().includes(value.toLowerCase())
            ),
    });
  };
